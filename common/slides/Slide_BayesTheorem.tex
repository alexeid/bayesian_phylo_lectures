% BAYES RULE
\begin{frame}
\frametitle{Bayes rule in statistics}

\LARGE{
\[
Pr(\theta|D) = \frac{Pr(D|\theta)Pr(\theta)}{Pr(D)} \quad %= \frac{Pr(D|\theta)Pr(\theta)}{\sum_{\theta}Pr(D|\theta)Pr(\theta)}
\]
%\begin{align*}
%Pr(\theta|D) &= \frac{Pr(D|\theta)Pr(\theta)}{\sum_{\theta}Pr(D|\theta)Pr(\theta)} \\
%&= \frac{Pr(D|\theta)Pr(\theta)}{Pr(D)}
%\end{align*}
}

\normalsize{
where 

\begin{itemize}
\item $P(D|\theta)$ is the \alert{likelihood}, 
\item $Pr(\theta)$ is the \alert{prior} distribution and
\item $Pr(\theta|D)$ is the \alert{posterior} distribution.
\item $Pr(D)$ is the \alert{marginal likelihood} of the data.
\end{itemize}
}

\end{frame}
